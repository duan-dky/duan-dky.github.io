//查找元素

std::pair<int, int> getElem(int a,list *l1)
{

    list *b=l1;
    int count=0;
    for(;b->next!=NULL;b=b->next){
        if(a==b->a){
            return std::pair<int, int>(a,count);
        }
        count++;
    }
    if(b->a==a){
        return std::pair<int, int>(a,count);
    }
    return std::pair<int, int>(a,-1);
    
}
